M2 增量更新测试
M2-A：FSEventsWatcher 功能验证（APFS/HFS+）
FSEventsWatcher 是库组件，当前 CLI 未直接暴露 watch 命令（属于 M4 UI 层使用）。可用以下快速 Swift 脚本验证：


cat > /tmp/testFSEvents.swift << 'EOF'
import Foundation
// 动态加载测试，需要将 IndexEngine 先 build 为动态库（M4 阶段）
// 此处通过手动创建文件，验证 FSEventStreamSetDispatchQueue 路径是否正确编译

// Quick compile check: 直接测试 FSEventStreamCreate API
import CoreServices

let paths = [NSHomeDirectory() + "/Desktop"] as CFArray
var ctx = FSEventStreamContext(version: 0, info: nil, retain: nil, release: nil, copyDescription: nil)
let flags = FSEventStreamCreateFlags(kFSEventStreamCreateFlagFileEvents | kFSEventStreamCreateFlagUseCFTypes)
let stream = FSEventStreamCreate(nil, { _, _, _, _, _, _ in
    print("FSEvents callback fired!")
}, &ctx, paths, FSEventStreamEventId(kFSEventStreamEventIdSinceNow), 1.0, flags)
print("Stream created:", stream != nil ? "OK" : "FAILED")
if let s = stream {
    FSEventStreamSetDispatchQueue(s, DispatchQueue.global())
    FSEventStreamStart(s)
    print("Watching ~/Desktop for 5 seconds...")
    Thread.sleep(forTimeInterval: 5)
    FSEventStreamStop(s)
    FSEventStreamInvalidate(s)
    FSEventStreamRelease(s)
}
EOF
swift /tmp/testFSEvents.swift
# 在另一个终端中: touch ~/Desktop/qs_test_$(date +%s).txt
# 预期：打印 "FSEvents callback fired!"
M2-B：DiskArbitrationMonitor 验证（外接盘热插拔）

cat > /tmp/testDA.swift << 'EOF'
import Foundation
import DiskArbitration

guard let session = DASessionCreate(kCFAllocatorDefault) else {
    print("FAILED: cannot create DA session")
    exit(1)
}
print("DA Session created: OK")

var info = "test_ctx"
DARegisterDiskAppearedCallback(session, nil, { disk, ctx in
    if let name = DADiskGetBSDName(disk) {
        print("Disk appeared:", String(cString: name))
    }
    if let desc = DADiskCopyDescription(disk) as? [String: AnyObject],
       let url = desc[kDADiskDescriptionVolumePathKey as String] as? URL {
        print("  Mount path:", url.path)
    }
}, nil)

DARegisterDiskDisappearedCallback(session, nil, { disk, _ in
    if let name = DADiskGetBSDName(disk) {
        print("Disk disappeared:", String(cString: name))
    }
}, nil)

DASessionSetDispatchQueue(session, DispatchQueue.main)
print("Monitoring disk events. Insert/remove an external drive...")
print("Press Ctrl+C to stop.")
RunLoop.main.run()
EOF
swift /tmp/testDA.swift
# 然后插入/拔出 U 盘或外接盘
# 预期：打印 "Disk appeared: diskX" / "Disk disappeared: diskX"
